Scroll Suave e Botão Subir para o Topo

    
- ex01 SOBE O SCROLL COM EFEITO SUAVE

function subirTela() {
  window.scrollTo({
    top: 0,
    left: 0,
    behavior: "smooth",
  });
}


- ex02 QUANDO O SCROLL FOR 0 (ESTIVER NO TOPO) BOTÃO SOME

function botaoVisivel() {
  if (window.scrollY === 0) {
    //ocultar botão
    document.querySelector(".scrollbutton").style.display = "none";
  } else {
    //mostrar botão
    document.querySelector(".scrollbutton").style.display = "block";
  }
}

window.addEventListener("scroll", botaoVisivel);
